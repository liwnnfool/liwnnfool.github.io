title: 理解闭包

```javascript

const outerValue = "Samurai";
const later;

function outerFunc() {
  const innerValue = "ninja";
  
  function innerFunc() {
    console.log(outerValue === "Samurai");
    console.log(innerValue === "ninja");
  }
  
  later = innerFunction;
}

outerFunction();

later();

```
